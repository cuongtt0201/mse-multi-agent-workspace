# ai_workspace_shared/base_agent.py — _execute_and_notify (rút gọn)
async def _execute_and_notify(self, task):
    plan_id, task_id = task.headers.plan_id, task.headers.task_id

    # 1. Memory Linking: đọc output_results các task upstream (needs_from)
    #    từ Blackboard HASH plan:{plan_id}:context — chỉ lấy task đã completed.
    task.body.params["dep_results"] = await self._gather_deps(task)

    # 2. Đánh dấu running rồi chạy domain logic (execute() là abstract).
    await set_task_status(self.redis_client, plan_id, task_id, "running")
    result = await self.execute(task)

    # 3. WRITE-BEFORE-PUBLISH: ghi output_results + completed lên Blackboard
    #    TRƯỚC khi báo downstream (đảm bảo dep sẵn sàng khi message tới).
    await set_task_output(self.redis_client, plan_id, task_id, result,
                          status="completed")

    # 4. Notify TRỰC TIẾP từng agent downstream qua routing key agent.{target}
    #    (KHÔNG có event "*.completed" broadcast — đẩy thẳng vào queue đích).
    for target in task.body.notify:        # vd "gmail:T4"
        await self.rmq.publish(f"agent.{target.split(':')[0]}", result_msg)
